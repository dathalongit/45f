# giá trị mặc định cho tham số
```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

    <pre>
        - giá trị mặc định cho tham số
        function TenHam(para1 = 1, para2 = 2) {
            // nội dung của hàm
        }
    </pre>

    <script>
        // gán giá trị mặc định cho tham số qua 1 dấu =
        function tinhTong(a = 1, b = 1) {
            console.log("a : " + a);
            console.log("b : " + b);
            var c;
            c = a + b;
            return c;
        }

        // gọi hàm 1 cách tự động
        // khi load trang html thì chúng gọi hàm

        // gọi hàm không truyền đối số
        var e = tinhTong();
        console.log("e là : " + e);

        // gọi hàm có truyền đối số
        var t = tinhTong(2,2);
        console.log("t là : " + t);
    </script>

</body>
</html>
```

```
            color: white;
            padding: 20px;

        }
    </style>

    <pre>
        lấy hay thay đổi giá trị của các thành phần trong form
    </pre>

    <div class="container">
        <div class="row">
            <label for="">A</label>
            <input type="text" name="a" value="">
        </div>
        <div class="row">
            <label for="">B</label>
            <input type="text" name="b" value="">
        </div>
        <div class="row">
            <label for="">Operator</label>
            <select name="operator" id="operator">
                <option value="+">+</option>
                <option value="-">-</option>
                <option value="/">/</option>
                <option value="*">*</option>
                <option value="%">%</option>
            </select>
        </div>

        <div class="row">
            <button id="print-btn" onclick="tinhToan()">In kết quả</button>
        </div>
    </div>

    <script>
        function tinhToan() {
            // .value lấy ra giá trị trong input
            var a = document.querySelectorAll("input[name='a']")[0].value;
            var b = document.querySelectorAll("input[name='b']")[0].value;

            /*
            * document.querySelectorAll("input[name='a']")[0].value = "giá trị mới";
            * */

            var c;

            var ope = document.querySelectorAll("select[name='operator']")[0].value;

            console.log(typeof a);
            console.log(typeof b);
            console.log(ope);

            // ép kiểu string về số
            a = parseFloat(a);
            b = parseFloat(b);

            if (ope == "+") {
                c = a + b;
            } else if (ope == "-") {
                c = a - b;
            } else if (ope == "*") {
                c = a * b;
            } else if (ope == "/") {
                c = a / b;
            } else {
                // % phép chia lấy dư
                c = a % b;
            }

            alert("kêt quả : " + c);
        }
    </script>

</body>
</html>
```

```
    </style>

<body>

    <pre style="font-size: 34px">
        js & style css
        truy cập css style
        cú pháp
        element.style.cssProperty
        vd :
        element.style.color, element.style.border

        quy tắc gán style css bằng js ( bỏ đi dấu - và từ thứ 2 biến chứ cái đầu tiên
        thành chữ viết hoa )

        background-color => backgroundColor
        margin-top => marginTop

    <button onclick="changeBackground('green')">Xanh</button>
    <button onclick="changeBackground('red')">Đỏ</button>
    <button onclick="changeBackground('violet')">Tím</button>
    <button onclick="changeBackground('orange')">Da cam</button>

    <button onclick="changeSize('+')">Phóng to</button>
    <button onclick="changeSize('-')">Thu nhỏ</button>

    <script>
        function changeBackground(color) {
            document.getElementById("target").style.backgroundColor = color;
        }

        function changeSize(size) {

            // lấy ra chiều rộng và chiều cao hiện tại
            var w = document.getElementById("target").clientWidth;
            var h = document.getElementById("target").clientHeight;

            //console.log(document.getElementById("target").style);
            // ép kiểu về số nguyên
            w = parseInt(w);
            h = parseInt(h);

            var ope;
            if (size == "+") {
                w = w + 100;
                h = h + 100;
            } else {
                w = w - 100;
                h = h - 100;
            }

            // nối thêm đơn vị px
            document.getElementById("target").style.width = w + "px";
            document.getElementById("target").style.height = h + "px";


        }
    </script>

</body>
</html>
```
