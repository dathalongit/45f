# dùng vòng lặp duyệt mảng
```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

    <pre>
        sử dụng vòng lặp duyệt mảng
    </pre>

    <ul id="list">

    </ul>

    <script>
        var cities = ["hà nội", "bắc ninh", "bắc giang", "hải phòng"];

        // duyệt mảng

        // tenMang.length độ dài của mảng
        var arrLength = cities.length;
        console.log(arrLength);

        // key trong mảng bắt đầu 0
        var html = "";
        for (var i = 0; i < arrLength; i++) {
             item = "<li>" + cities[i] + "</li>";
             console.log(item);
             html = html + item;
        }
        console.log(html);
        document.getElementById("list").innerHTML = html;
    </script>

</body>
</html>
```

# tính toán thời gian
```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

    <pre>
        tính toán trong js
        getTime() trả về số mili giây tính 1.1.1970
        1s = 1000ms
    </pre>

    <script>
        // tinh xem còn bao lâu đến tết âm lịch
        var d1 = new Date();
        var d1ms = d1.getTime();
        document.writeln("<br>" + d1ms);

        var d2 = new Date(2021, 1, 12);
        var d2ms = d2.getTime();
        document.writeln("<br>" + d2ms);

        var distance = d2ms - d1ms;
        document.writeln("<br> mili giây : " + distance);
        var distanceS = distance/1000;
        document.writeln("<br> giây : " + distanceS);
        var distanceD = distanceS/(24*60*60);
        document.writeln("<br> ngày : " + distanceD);



    </script>

</body>
</html>
```

# phạm vi của biến
```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

    <pre>
        variable scope ( phạm vi của biến )
        - phạm vi toàn cục ( global )
        - phạm vi cục bộ ( local )

    </pre>

    <script>
        // khai báo các biến bên ngoài hàm gọi là biến toàn cục
        var d = new Date();


        // code here can NOT use carName

        function myFunction() {

            // khởi tạo và khai báo từ khóa var trong function
            // thì biến cục bộ
            var carName = "Volvo";

            // code here CAN use carName

            console.log("trong hàm : " + carName);
            console.log("trong hàm : " + d);

        }
        myFunction();


        console.log("ngoài hàm " + d);

        console.log("ngoài hàm : " + carName);
    </script>

</body>
</html>
```
