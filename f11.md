# dùng vòng lặp duyệt mảng
```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

    <pre>
        sử dụng vòng lặp duyệt mảng
    </pre>

    <ul id="list">

    </ul>

    <script>
        var cities = ["hà nội", "bắc ninh", "bắc giang", "hải phòng"];

        // duyệt mảng

        // tenMang.length độ dài của mảng
        var arrLength = cities.length;
        console.log(arrLength);

        // key trong mảng bắt đầu 0
        var html = "";
        for (var i = 0; i < arrLength; i++) {
             item = "<li>" + cities[i] + "</li>";
             console.log(item);
             html = html + item;
        }
        console.log(html);
        document.getElementById("list").innerHTML = html;
    </script>

</body>
</html>
```

```
        var distance = d2ms - d1ms;
        document.writeln("<br> mili giây : " + distance);
        var distanceS = distance/1000;
        document.writeln("<br> giây : " + distanceS);
        var distanceD = distanceS/(24*60*60);
        document.writeln("<br> ngày : " + distanceD);



    </script>

</body>
</html>
```

# phạm vi của biến
```
<!DOCTYPE html>

    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

    <pre>
        variable scope ( phạm vi của biến )
        - phạm vi toàn cục ( global )
        - phạm vi cục bộ ( local )

    </pre>

    <script>
        // khai báo các biến bên ngoài hàm gọi là biến toàn cục
        var d = new Date();


        // code here can NOT use carName

        function myFunction() {

            // khởi tạo và khai báo từ khóa var trong function
            // thì biến cục bộ
            var carName = "Volvo";

            // code here CAN use carName

            console.log("trong hàm : " + carName);
            console.log("trong hàm : " + d);

        }
        myFunction();


        console.log("ngoài hàm " + d);

        console.log("ngoài hàm : " + carName);
    </script>

</body>
</html>
```

# object ( đối tượng ) trong js
```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

    <pre>
        object ( đối tượng ) trong js

    </pre>

    <script>
        var car = {type:"Fiat", model:"500", color:"white", price: 20000000};

        var person = {firstName:"John", lastName:"Doe", age:50, eyeColor:"blue"};

        // truy cập property object
        document.writeln("<br>" + car.price);
        document.writeln("<br>" + car["price"]);

        car.price = 3000000000;

        // vừa có property vừa có method
        var person2 = {
            firstName: "John",
            lastName : "Doe",
            id       : 5566,
            diemToan: 10,
            diemLy:7,
            diemHoa: 8,
            diemTb: 0,
            // cách tạo 1 method
            fullName : function() {
                // trong method của object thì từ khóa this đại diện cho chính bản thân nó
                return this.firstName + " " + this.lastName;
            },
            diemTB: function() {
                this.diemTb = ((this.diemHoa+this.diemLy+this.diemToan)/3);
                return this.diemTb;
            },
            result: function() {
                this.diemTb = ((this.diemHoa+this.diemLy+this.diemToan)/3);
                if (this.diemTb > 5) {
                    return "đỗ";
                }
                return "trượt";
            }
        };

        console.log(person2.diemToan);
        console.log(person2["diemToan"]);
        console.log(person2.diemTB());
        console.log(person2.result());




    </script>
</body>
</html>
```
