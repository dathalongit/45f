## mã nguồn
```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

    <script>

        var tuoi = 21;
        var message;
        var message2;
        message = (tuoi >= 18) ? "người trưởng thành" : "trẻ em";
        console.log(message);

        if (tuoi >= 18) {
            message2 = "người trưởng thành";
        } else {
            message2 = "trẻ em";
        }
        console.log(message2);
    </script>

</body>
</html>
```
# cú pháp switch ... case
![image](https://user-images.githubusercontent.com/50043998/162605152-24e2dec2-83ed-4936-a9bf-238974db60f4.png)
```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

    <script>
        var thu = 5;
        // in thời khóa biểu
        switch (thu) {
            case 2:
               console.log("thứ 2 học html");
               break;
            case 3:
                console.log("thứ 3 học css");
                break;
            case 4:
                console.log("thứ 4 học js");
                break;
            case 5:
                console.log("thứ 5 học php");
                break;
            case 6:
                console.log("thứ 6 học mysql");
                break;
            case 7:
                console.log("thứ 7 học laravel");
                break;
            default:
                console.log("chủ nhật nghỉ ngơi");
        }

    </script>

</body>
</html>
```
