```
        // in thời khóa biểu

               console.log("thứ 2 học html");
               break;
            case 3:
                console.log("thứ 3 học css");
                break;
            case 4:
                console.log("thứ 4 học js");
                break;
            case 5:
                console.log("thứ 5 học php");
                break;
            case 6:
                console.log("thứ 6 học mysql");
                break;
            case 7:
                console.log("thứ 7 học laravel");
                break;
            default:
                console.log("chủ nhật nghỉ ngơi");
        }

    </script>

</body>
</html>
```

```
    In ra bàn cờ vua js trong thẻ có #bancovuajs với 8*8 = 64 ô

    <div class="cell cell1" style="top: 0; left: 0; background: greenyellow;">
        cell 1
    </div>

</div>

<script>
    var html = "";
    var topCss = 0;
    var leftCss = 0;
    var backgroundCss = "greenyellow";
    // hàng
    var row = 0;

    for (i = 1; i<= 64;i++) {

        leftCss = leftCss + 100;
        // 1,9,17, 25,33,41... ô đầu tiên trong hàng ngang
        if (((i-1) % 8) == 0) {
            // ô đầu tiên trong hàng
            leftCss = 0;
        }

        // đổi màu
        if (row % 2 == 0) {
            // hàng chẵn
            if ( i % 2 != 0) {
                backgroundCss = "orange";
            } else {
                backgroundCss = "greenyellow";
            }
        } else {
            // hàng lẽ
            if ( i % 2 != 0) {
                backgroundCss = "greenyellow";
            } else {
                backgroundCss = "orange";
            }
        }


        html = html + "<div class=\"cell cell"+i+"\" style=\"top: "+topCss+"px; left: "+leftCss+"px; background: "+backgroundCss+";\">cell "+i+"</div>";

        if ( (i % 8) == 0 )  {
            // bắt đầu chu kỳ hàng mới
            leftCss = 0;
            //
            topCss = topCss + 100;
            // hàng
            row++;
        }
    }
    document.getElementById("bancovuajs").innerHTML = html;

</script>
</body>
</html>
```
